function setupPresets() {
    const container = document.getElementById(&#39;preset-container&#39;);
    Object.keys(PRESETS).forEach(key =&gt; {
        const btn = document.createElement(&#39;div&#39;);
        btn.className = &#39;preset-btn&#39;;
        btn.innerText = PRESETS[key].name;
        btn.onclick = () =&gt; applyPreset(key);
        btn.id = `btn-preset-${key}`;
        container.appendChild(btn);
    });
}

function syncControls() {
    const params = [&#39;temp&#39;, &#39;topp&#39;, &#39;topk&#39;, &#39;tokens&#39;];
    params.forEach(p =&gt; {
        const slider = document.getElementById(`param-${p}`);
        const display = document.getElementById(`val-${p}`);
        slider.oninput = () =&gt; { display.innerText = slider.value; };
    });
}

function applyPreset(key) {
    const p = PRESETS[key];
    if (!p) return;

    document.querySelectorAll(&#39;.preset-btn&#39;).forEach(b =&gt; b.classList.remove(&#39;active&#39;));
    document.getElementById(`btn-preset-${key}`).classList.add(&#39;active&#39;);

    document.getElementById(&#39;param-temp&#39;).value = p.temp;
    document.getElementById(&#39;param-topp&#39;).value = p.topp;
    document.getElementById(&#39;param-topk&#39;).value = p.topk;
    document.getElementById(&#39;system-prompt&#39;).value = p.system;

    // Trigger visual sync
    [&#39;temp&#39;, &#39;topp&#39;, &#39;topk&#39;].forEach(id =&gt; {
        document.getElementById(`val-${id}`).innerText = p[id];
    });
}

// --- API LOGIC ---
async function validateKey() {
    const keyInput = document.getElementById(&#39;api-key&#39;);
    const btn = document.getElementById(&#39;btn-validate&#39;);
    const status = document.getElementById(&#39;status-msg&#39;);
    
    API_KEY = keyInput.value.trim();
    if (!API_KEY) {
        showStatus(&quot;Please enter an API key first.&quot;, &quot;err&quot;);
        return;
    }

    btn.disabled = true;
    btn.innerHTML = &#39;&lt;span class=&quot;loader&quot;&gt;&lt;/span&gt; Validating...&#39;;
    
    try {
        // Test call to list models
        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models?key=${API_KEY}`);
        const data = await response.json();

        if (response.ok) {
            showStatus(&quot;✅ API Key Valid! Models loaded.&quot;, &quot;ok&quot;);
            populateModels(data.models);
            document.getElementById(&#39;btn-run&#39;).disabled = false;
        } else {
            showStatus(`❌ Error: ${data.error?.message || &quot;Invalid Key&quot;}`, &quot;err&quot;);
        }
    } catch (e) {
        showStatus(&quot;❌ Network error. Check your internet connection.&quot;, &quot;err&quot;);
    } finally {
        btn.disabled = false;
        btn.innerText = &quot;Validate API Key&quot;;
    }
}

function populateModels(models) {
    const select = document.getElementById(&#39;model-select&#39;);
    select.innerHTML = &quot;&quot;;
    select.disabled = false;

    // Filter for generative models
    const filtered = models.filter(m =&gt; m.supportedGenerationMethods.includes(&quot;generateContent&quot;));
    
    filtered.forEach(m =&gt; {
        const opt = document.createElement(&#39;option&#39;);
        opt.value = m.name; // e.g. &quot;models/gemini-pro&quot;
        opt.innerText = m.displayName;
        if (m.name.includes(&quot;gemini-1.5-flash&quot;)) opt.selected = true;
        select.appendChild(opt);
    });
}

function showStatus(msg, type) {
    const el = document.getElementById(&#39;status-msg&#39;);
    el.innerText = msg;
    el.className = type === &#39;ok&#39; ? &#39;status-ok&#39; : &#39;status-err&#39;;
    el.style.display = &#39;block&#39;;
}

async function runInference() {
    const outArea = document.getElementById(&#39;output-area&#39;);
    const metaArea = document.getElementById(&#39;output-meta&#39;);
    const btn = document.getElementById(&#39;btn-run&#39;);
    const userPrompt = document.getElementById(&#39;user-prompt&#39;).value.trim();
    const systemPrompt = document.getElementById(&#39;system-prompt&#39;).value.trim();

    if (!userPrompt) {
        outArea.innerText = &quot;Please enter a question first!&quot;;
        return;
    }

    btn.disabled = true;
    outArea.innerText = &quot;Thinking...&quot;;
    metaArea.innerText = &quot;&quot;;
    const startTime = Date.now();

    const config = {
        temperature: parseFloat(document.getElementById(&#39;param-temp&#39;).value),
        topP: parseFloat(document.getElementById(&#39;param-topp&#39;).value),
        topK: parseInt(document.getElementById(&#39;param-topk&#39;).value),
        maxOutputTokens: parseInt(document.getElementById(&#39;param-tokens&#39;).value)
    };

    const model = document.getElementById(&#39;model-select&#39;).value;
    
    try {
        const result = await callGemini(model, systemPrompt, userPrompt, config);
        const duration = ((Date.now() - startTime) / 1000).toFixed(2);
        
        outArea.innerText = result.text;
        metaArea.innerText = `Model: ${result.modelUsed} | Latency: ${duration}s | Status: Success`;
    } catch (err) {
        outArea.innerText = `Error: ${err.message}`;
        metaArea.innerText = &quot;Try adjusting your settings or checking your API quota.&quot;;
    } finally {
        btn.disabled = false;
    }
}

async function callGemini(modelPath, system, user, config) {
    // Fallback logic
    const modelsToTry = [modelPath, &quot;models/gemini-1.5-flash&quot;, &quot;models/gemini-pro&quot;];
    let lastError = &quot;&quot;;

    for (let m of modelsToTry) {
        try {
            const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/${m}:generateContent?key=${API_KEY}`, {
                method: &#39;POST&#39;,
                headers: { &#39;Content-Type&#39;: &#39;application/json&#39; },
                body: JSON.stringify({
                    contents: [
                        { role: &quot;user&quot;, parts: [{ text: `Instruction: ${system}\n\nQuestion: ${user}` }] }
                    ],
                    generationConfig: config
                })
            });

            const data = await response.json();
            if (response.ok) {
                return {
                    text: data.candidates[0].content.parts[0].text,
                    modelUsed: m
                };
            } else {
                lastError = data.error?.message || &quot;Unknown error&quot;;
            }
        } catch (e) {
            lastError = e.message;
        }
    }
    throw new Error(lastError);
}
